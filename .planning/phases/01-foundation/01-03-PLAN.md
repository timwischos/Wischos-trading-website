**src/components/layout/SiteHeader.tsx**

Desktop: horizontal nav bar with logo on the left, nav links in the center/right, and a primary CTA button. Mobile (below md breakpoint): hide nav links, show hamburger icon that opens a shadcn Sheet drawer with the same links.

Requirements:
- Import `navigation` from `~/content/navigation`
- Use TanStack Router `<Link>` for all navigation links (enables active state detection)
- Desktop nav: links visible at `md:flex`, hidden on mobile
- Mobile nav: `Sheet` component from shadcn/ui, trigger is a `Menu` icon from lucide-react
- CTA button uses shadcn `Button` with `variant="default"` — links to `navigation.cta.href`
- Logo text from `navigation.logoText` — link to "/"
- Sheet closes when a nav link is clicked (use `SheetClose` wrapper or `useState` to control open state)
- Use semantic `<header>` and `<nav>` elements
- Tailwind classes for styling: sticky top, white background, border-b, z-50

```typescript
'use client'  // Sheet requires client interactivity

import { useState } from 'react'
import { Link } from '@tanstack/react-router'
import { Menu } from 'lucide-react'
import { Button } from '~/components/ui/button'
import { Sheet, SheetContent, SheetTrigger } from '~/components/ui/sheet'
import { navigation } from '~/content/navigation'

export function SiteHeader() {
  const [mobileOpen, setMobileOpen] = useState(false)

  return (
    <header className="sticky top-0 z-50 w-full border-b bg-background/95 backdrop-blur">
      <div className="container mx-auto flex h-16 items-center justify-between px-4">
        {/* Logo */}
        <Link to="/" className="font-semibold text-lg">
          {navigation.logoText}
        </Link>

        {/* Desktop navigation */}
        <nav className="hidden md:flex items-center gap-6">
          {navigation.links.map((link) => (
            <Link
              key={link.href}
              to={link.href}
              className="text-sm font-medium text-muted-foreground hover:text-foreground transition-colors"
              activeProps={{ className: 'text-foreground' }}
            >
              {link.label}
            </Link>
          ))}
        </nav>

        {/* Desktop CTA */}
        <div className="hidden md:flex">
          <Button asChild>
            <Link to={navigation.cta.href}>{navigation.cta.label}</Link>
          </Button>
        </div>

        {/* Mobile navigation */}
        <Sheet open={mobileOpen} onOpenChange={setMobileOpen}>
          <SheetTrigger asChild className="md:hidden">
            <Button variant="ghost" size="icon" aria-label="Open navigation">
              <Menu className="h-5 w-5" />
            </Button>
          </SheetTrigger>
          <SheetContent side="right" className="w-64">
            <nav className="flex flex-col gap-4 pt-8">
              {navigation.links.map((link) => (
                <Link
                  key={link.href}
                  to={link.href}
                  className="text-base font-medium"
                  onClick={() => setMobileOpen(false)}
                >
                  {link.label}
                </Link>
              ))}
              <Button asChild className="mt-4">
                <Link to={navigation.cta.href} onClick={() => setMobileOpen(false)}>
                  {navigation.cta.label}
                </Link>
              </Button>
            </nav>
          </SheetContent>
        </Sheet>
      </div>
    </header>
  )
}
```

import {
  createRootRoute,
  HeadContent,
  Outlet,
  Scripts,
} from '@tanstack/react-router'
import { SiteHeader } from '~/components/layout/SiteHeader'
import { SiteFooter } from '~/components/layout/SiteFooter'
import { siteMeta } from '~/content/meta'
import '~/styles/globals.css'

export const Route = createRootRoute({
  head: () => ({
    meta: [
      { charSet: 'utf-8' },
      { name: 'viewport', content: 'width=device-width, initial-scale=1' },
      { title: siteMeta.defaultTitle },
      { name: 'description', content: siteMeta.defaultDescription },
    ],
    links: [
      { rel: 'icon', href: '/favicon.ico' },
    ],
  }),
  component: RootComponent,
})

function RootComponent() {
  return (
    <html lang="en">
      <head>
        <HeadContent />
      </head>
      <body className="min-h-screen bg-background font-sans antialiased">
        <SiteHeader />
        <main>
          <Outlet />
        </main>
        <SiteFooter />
        <Scripts />

}
